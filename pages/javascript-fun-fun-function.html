1.1. bind and this

// basic object creation
let dog = {
  sound: 'woof',
  talk: function() {
     console.log(this.sound);
  }
}

dog.talk() // returns: woof

// 'this' points to nothing
// in JS 'this' refers to the context where it was called
// (vs the context where it was defined)
let talkFunction = dog.talk;
talkFunction(); // returns: undefined

// solve above problem by binding to an object
// here: bind 'dog' to 'this'
let boundFunction = talkFunction.bind(dog);
boundFunction(); // returns: woof
// real life example:
let button = document.getElementById('myNiceButton');
button.addEventListener('click', dog.talk); // -> 'this' would refer to the button
// solution:
button.addEventListener('click', dog.talk.bind(dog));

1.2. Examples of this and bind

// rather than doing:
function talk1(sound) {
  console.log(sound);
}

talk1('wooof'); // returns: wooof

// do below:
function talk2() {
  console.log(this.sound);
}

talk2(); // <- 'this' refers to the global object



// 'this' refers to bound object; here: boromir
let boromir = {
  speak: talk2,
  sound: "One does not simply walk into mordor"
}
let talkBoundToBoromir = talk2.bind(boromir);
talkBoundToBoromir() // returns: One does not simply walk into mordor



// 'this' refers to boromir
boromir.speak(); // <- 'talk2' assigned to a property



// 'this' refers to global object
let blabber = boromir.speak;
blabber(); // returns: undefined

// solution:
boromir = {
  sound: "One does not simply walk into mordor"
}
// make new bound function
boromir.speak = talk2.bind(boromir); // <- bind to boromir object
blabber = boromir.speak;
blabber(); // returns One does not simply walk into mordor



// 'this' is context-sensitive
// (Don't write this kind of code -> too confusing)
boromir = {
  blabber: talk2,
  sound: "One does not simply walk into mordor"
}
let gollum = {
  jabber: boromir.blabber,
  sound: "My preciousssssss..."
}
gollum.jabber(); // returns: My preciousssssss...

1.3. Prototype basics

  • Inheritance in JS is achieved through prototypes
  • Prototypes do not create a copy of the original object, but a delegate to it
function talk() {
  console.log(this.sound);
}

// prototype:
let animal = {
  // talk: talk
  talk // ES6: same as above
}
animal.talk(); // undefined

let cat = {
  sound: 'meow!'
}

let dog = {
  sound: 'woof!'
}

Object.setPrototypeOf(cat, animal); // set prototype of cat to be animal
Object.setPrototypeOf(dog, animal);
cat.talk(); // meow!
dog.talk(); // woof!



// set a prototype chain
let prarieDog = {
  howl: function() {
    console.log(this.sound.toUpperCase())
  }
}

Object.setPrototypeOf(prarieDog, dog);
prarieDog.howl(); // WOOF!



// prototypes create delegates
Object.setPrototypeOf(dog, animal);
animal.talk = function() {
  console.log('i am a little teapot');
}
dog.talk(); // i am a little teapot

1.4. The new keyword

  • here: new applied to functions (not applied to classes)
// Constructor
function Person(saying) {
  this.saying = saying;
}

Person.prototype.talk = function(){
  console.log('I say:', this.saying)
};

var crockforg = new Person('SEMICOLANS!!!1one1');
crockforg.talk();

What does new do?

  1. Creates a new object
  2. Looks at whatever we have called new on (Person = Constructor), checks that object's prototype property (Person.prototype), and sets the prototype of the new object to be that
  3. Looks again at the constructor (Person), and calls the function in the object created in step 1 assigned to the this variable (this.saying)
  4. Returns the object
function mynew(constructor) {
  // Step 1: create object
  var obj = {};

  // Step 2: set prototype
  Object.setPrototypeOf(obj, constructor.prototype)

  // Step 3: execute constructor with 'this'
  // var argsArray = Array.from(arguments); // ES6
  var argsArray = Array.prototype.slice.apply(arguments) // ES5
  constructor.apply(obj, argsArray.slice(1));

  // Step 4: return the created objects
  return obj;
}

var crockforg = mynew(Person, 'SEMICOLANS!!!1one1');
crockforg.talk();
function Person(saying) {
  this.saying = saying;
  return {
    dumbObject: true
  }
}

Person.prototype.talk = function(){
  console.log('I say:', this.saying)
};

// CORRECT IMPLEMENTATION
function mynew(constructor) {
  // Step 1: create object
  var obj = {};

  // Step 2: set prototype
  Object.setPrototypeOf(obj, constructor.prototype)

  // Step 3: execute constructor with 'this'
  // var argsArray = Array.from(arguments); // ES6
  var argsArray = Array.prototype.slice.apply(arguments) // ES5
  
  // Step 4: return the created objects
  return constructor.apply(obj, argsArray.slice(1)) || obj;
}

var crockforg = mynew(Person, 'SEMICOLANS!!!1one1');
console.log(crockforg); // { dumbObject: true }

1.5. __proto__ vs prototype

1.5.1. __proto__

  • __proto__ is the property on an object that points to the prototype that has been set for that object (OBJECT DELEGATION)
  • Example: borrow pen from friend's friend's friend
let cat = { breed: 'munchkin' }
let myCat = { name: 'Fluffykins' }
Object.setPrototypeOf(myCat, cat);

console.log(myCat.name, myCat.breed); // returns: Fluffykins munchkin

console.log(myCat.__proto__); // returns: { breed: 'munchkin' }

// __proto__ is a REFERENCE to the object, NOT an INSTANCE
cat.tailLength = 14;
console.log(myCat.__proto__); // returns: { breed: 'munchkin', tailLength: 14 }

1.5.2. prototype

  • prototype is a property on a function and it is set if you're using the new keyword
// Constructor
function Dog() {}

Dog.prototype.breed = 'Bulldog'; // 'prototype' can be ONLY created on functions

let myDoggie = new Dog();

console.log(myDoggie.prototype); // no prototype on myDoggie // returns: undefined
console.log(myDoggie.__proto__); // refers to prototype of Dog // returns: Dog { breed: 'Bulldog' }
console.log(Dog.prototype) // however there is a prototype on Dog // returns: Dog { breed: 'Bulldog' }



function Giraffe() {} // constructor function
console.log(Giraffe.prototype) // has a prototype property // returns: Giraffe {}

let koala = {} // simple object
console.log(koala.prototype) // has NO prototype property // returns: undefined
console.log(koala.__proto__) // but has a __proto__ // returns: {}
console.log(koala.__proto__ === Object.prototype ) // -> refers to the global object prototype // returns: true

Object.prototype.waffle = 'hohoho';
console.log(koala.waffle); // returns: hohoho
// this means 'Object' is acutally a function ('prototype' can be ONLY created on functions)
console.log(Object()) // returns: {}

1.6. Object.create()

1.6.1. What is Object.create()

Object.create() is a static method that creates a new object with the prototype set to a certain object

const cat = {
  init: function(sound) {
    this.sound = sound;
  },
  makeSound: function() {
    console.log(this.sound);
  }
}

const mark = Object.create(cat);
mark.init('mewwwww'); // gets the property from prototype of cat
mark.makeSound(); // gets the property from prototype of cat // returns: mewwwww

const waffles = Object.create(cat);
waffles.init('mrrrrowww'); // gets the property from prototype of cat
waffles.makeSound(); // gets the property from prototype of cat // returns: mrrrrowww

console.log('Is mark a cat?', cat.isPrototypeOf(mark)); // returns true

Replicating Object.create()

const cat = {
  makeSound: function() {
    console.log(this.sound)
  }
}

function objectCreate(proto) {
  const obj = {};
  Object.setPrototypeOf(obj, proto);
  return obj;
}

const waffles = objectCreate(cat);
waffles.sound = 'mrrrrowww';
waffles.makeSound(); // returns: mrrrowww

1.7. class keyword

  • class is just a very thin layer on top of the prototyping inheritance (has no real class-based inheritance as in other languages)
class Mammal {
  // no private members in JS (as in other programming languages)
  // convention is to add _ to indicate the member is private
  constructor(sound) {
    this._sound = sound;
  }

  talk() {
    return this._sound;
  }
}

// INHERITANCE
class Dog extends Mammal {
  constructor() {
    super('woffeliwofffff'); // calls the constructor of inherited class
  }
}

let fluffykins = new Mammal('woof!');
console.log(fluffykins); // returns: Mammal { sound: 'woof!' }
console.log(fluffykins.talk()); // returns: woof!

let doggy = new Dog();
console.log(doggy.talk()); // returns: woffeliwofffff

// can access variables of a class from outside
console.log(fluffykins._sound); // returns: woof!
// can even change it
fluffykins._sound = 1;
console.log(fluffykins._sound); // returns: 1



// JS has no real `class` (it is just an alias for a constructor)
console.log(typeof Dog); // returns: function

let x = Dog.prototype.talk.bind({
  _sound: 'ROAR'
})();
console.log(x); // returns: ROAR

// can even do this
let y = Dog.prototype.isPrototypeOf(doggy);
console.log(y); // returns: true

2. Factory Functions

https://medium.com/humans-create-software/factory-functions-in-javascript-video-d38e49802555

  • Factories are simply functions that create objects.
  • You can use factories instead of using classes, and factories are simpler than classes and easier to reason about.
class Dog {
  constructor() {
    this.sound = 'woof'
  }
  talk() {
    console.log(this.sound)
  }
}
const sniffles = new Dog()
sniffles.talk() // Outputs: "woof"

// Problem:
$('button.myButton').click(sniffles.talk) // <-- this will break, because 'this' points to the button

// Solution:
// Method (1): use bind on sniffles
$('button.myButton').click(sniffles.talk.bind(sniffles))
// OR Method (2): wrap the talk call in a function
$('button.myButton').click( _ => sniffles.talk() )

// however syntax is very unintuitive
// better way is to use FACTORY FUNCTIONS

const dog = () => {
  const sound = 'woof'; // <-- simple variable
  return { // <-- returns a simple object literal with one property
    talk: () => console.log(sound)
    // because of the awesomeness of closures, 'talk' has access to the variable 'sound'
    // code outside of 'dog' has no access to it (basically makes it private)
  }
};

// we can use simply
$('button.myButton').click(sniffles.talk)
  • However use classes, when you are creating a lot of objects, because classes have a better performance
    • Using a factory takes around 0.0004 milliseconds, while using a class takes 0.0002 milliseconds.
    • This means that if you are creating 10,000 instances, using a class will earn you 2ms over factories, which isnt going to matter. If youre creating 10,000 objects or less, you should stick to factories.
    • If youre creating 100,000 instances per second, classes might be for you, but you should really ask yourself if the problem is that youre creating a hundred thousance instances per second in the first place.

3. Composition over Inheritance

https://www.youtube.com/watch?v=wfMtDGfHWpA&list=PL0zVEGEvSaeGPBRt-y2QZ3wh64XAe40jx&index=2

  • Inheritance is when you design your types around what they are
  • Composition is when you design your types around what they do

Inheritance

Robot
  .drive()

  MurderRobot
    .kill()

  CleaningRobot
    .clean()

Animal
  .poop()

  Dog
    .bark()

  Cat
    .meow()

// What to do, when we need a MurderRobotDog (don't need '.poop()')
// --> doesn't fit into this inheritance hierarchy

Composition

dog            = pooper + barker;
cat            = pooper + meower;
cleaningRobo   = driver + cleaner;
murderRobot    = driver + killer;
murderRobotDog = driver + killer + barker;

const barker = (state) => ({
  bark: () => console.log('Woof, I am ' + state.name)
});

const driver = (state) => ({
  drive: () => state.position = state.postion + state.speed
});

barker({name: 'karo'}).bark() // Woof, I am karo



// factory function (ES6)
const murderRobotDog = (name) => {
  let state = {
    name,
    speed: 100,
    postion: 0
  }
  return Object.assign(
    {},
    barker(state),
    driver(state),
    killer(state)
  )
}

murderRobotDog('sniffles').bark() // Woof, I am sniffles
  • When to use composition/inheritance?
    • most developers say you should favor composition over inheritance
    • basic guidance:
      • 'is-a' relationship --> use inheritance (eg. Mike 'is-a' Person)
      • 'has-a' relationship --> use composition (eg. Car 'has-a(n)' Engine)
